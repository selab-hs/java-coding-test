# DNA 비밀번호
* 시간 초과로 답지 확인
* 코드를 작성하는데 계속해서 틀렸다고 나와서 답지를 확인함

```java
import java.io.*;
import java.util.*;

class Main {
    static int checkArr[];
    static int myArr[];
    static int checkSecret;

    public static void main(String[] args) throws NumberFormatException, IOException {
        BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(bf.readLine());
        int S = Integer.parseInt(st.nextToken());   //문자열의 길이
        int P = Integer.parseInt(st.nextToken());   //비밀번호의 길이
        int Result = 0;                             //결과
        char A[];
        checkArr = new int[4];
        myArr = new int[4];
        checkSecret = 0;
        A = bf.readLine().toCharArray();            //문자열을 받아서 A에 char 배열로 저장
        st = new StringTokenizer(bf.readLine());
        for (int i = 0; i < 4; i++) {
            checkArr[i] = Integer.parseInt(st.nextToken());
            if (checkArr[i] == 0) checkSecret++;    //만약 비밀번호 유효성 여부에 관여하지 않는 문자가 있을 시에는 checkSecret++;
        }

        for (int i = 0; i < P; i++) { //초기 P부분 문자열 처리 부분
            add(A[i]);
        }

        if (checkSecret == 4) Result++; //만약 모든 문자열이 관여하지 않는 경우에는 result에 1을 더한다
        // 슬라이딩 윈도우 처리 부분
        for (int i = P; i < S; i++) {       // i : 끝 포인터 , j : 시작 포인터
            int j = i - P;
            add(A[i]);
            remove(A[j]);
            if (checkSecret == 4) Result++;  // 4자리 수에 대한 크기가 모두 충족되었을 때는 유효한 비밀번호
        }
        System.out.println(Result);
        bf.close();
    }

    private static void add(char c) { //새로 들어온 문자를 처리해주는 함수
        switch (c) {
            case 'A':
                myArr[0]++;
                if (myArr[0] == checkArr[0])
                    checkSecret++;
                break;
            case 'C':
                myArr[1]++;
                if (myArr[1] == checkArr[1])
                    checkSecret++;
                break;
            case 'G':
                myArr[2]++;
                if (myArr[2] == checkArr[2])
                    checkSecret++;
                break;
            case 'T':
                myArr[3]++;
                if (myArr[3] == checkArr[3])
                    checkSecret++;
                break;
        }
    }

    private static void remove(char c) { //제거되는  문자를 처리해주는 함수
        switch (c) {
            case 'A':
                if (myArr[0] == checkArr[0])
                    checkSecret--;
                myArr[0]--;
                break;
            case 'C':
                if (myArr[1] == checkArr[1])
                    checkSecret--;
                myArr[1]--;
                break;
            case 'G':
                if (myArr[2] == checkArr[2])
                    checkSecret--;
                myArr[2]--;
                break;
            case 'T':
                if (myArr[3] == checkArr[3])
                    checkSecret--;
                myArr[3]--;
                break;
        }
    }
}
```

# 배운 것
기존에 내가 썼던 코드는 new int[4] 형태의 비교하기 위한 배열을 하나 가지고 있었다. 이를 통해서 새로 들어오는 문자와 나가는 문자를 비교하면서 계산했는데 이 방법은 계산할 수는 있지만 위의 코드보다는 시간이 오래 걸리는 방법이었다.
그리고 답안 코드를 보면서 생각난 것인데
```java
import java.io.*;
import java.util.*;

class Main {
    private static int[] cal = new int[4];
    private static int start = 0;
    private static int end;
    private static String password;

    public static void main(String[] args) throws IOException {
        BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer stringTokenizer = new StringTokenizer(bufferedReader.readLine());

        int stringLength = Integer.parseInt(stringTokenizer.nextToken());
        int passwordLength = Integer.parseInt(stringTokenizer.nextToken());
        end = passwordLength - 1;

        stringTokenizer = new StringTokenizer(bufferedReader.readLine());
        password = stringTokenizer.nextToken();

        stringTokenizer = new StringTokenizer(bufferedReader.readLine());
        int[] condition = new int[4];

        for (int i = 0; i < 4; i++) {
            condition[i] = Integer.parseInt(stringTokenizer.nextToken());
        }

        int count = 0;

        //초기 계산 배열 초기화
        for (int i = 0; i < passwordLength; i++) {
            switch (password.charAt(i)) {
                case 'A':
                    cal[0]++;
                    break;
                case 'C':
                    cal[1]++;
                    break;
                case 'G':
                    cal[2]++;
                    break;
                case 'T':
                    cal[3]++;
                    break;
                default:
                    break;
            }
        }

        for (int i = 0; i <= stringLength - passwordLength; i++) {
            if (Arrays.equals(cal, condition)) count++;
            if(i != stringLength-passwordLength) slide();
        }

        System.out.println(count);
    }

    private static void slide() {
        switch (password.charAt(start)) {
            case 'A':
                cal[0]--;
                break;
            case 'C':
                cal[1]--;
                break;
            case 'G':
                cal[2]--;
                break;
            case 'T':
                cal[3]--;
                break;
            default:
                break;
        }

        switch (password.charAt(end + 1)) {
            case 'A':
                cal[0]++;
                break;
            case 'C':
                cal[1]++;
                break;
            case 'G':
                cal[2]++;
                break;
            case 'T':
                cal[3]++;
                break;
            default:
                break;
        }
        start++;
        end++;
    }
}
```
내 코드를 보면 `if (Arrays.equals(cal, condition)) count++;`부분이 있는데 이는 잘못된 조건문이다. 같아야하는 것이 아니라 최소 조건이기 때문에 다시 비교해야 한다. 이 부분을 `if (cal[0] >= condition[0] && cal[1] >= condition[1] && cal[2] >= condition[2] && cal[3] >= condition[3]) count++;`으로 수정하니 정답 처리 되었다.
* 조건을 제대로 읽어보자
