# 스택 수열
* 시간 초과로 답지 확인

```java
import java.util.Scanner;
import java.util.Stack;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        int[]A = new int[N];
        for (int i = 0; i < N; i++) {
            A[i] = sc.nextInt();
        }
        Stack<Integer> stack = new Stack<>();
        StringBuffer stringBuffer = new StringBuffer();
        int num = 1; // 오름차순 수
        boolean result = true;
        for (int i = 0; i < A.length; i++) {
            int currSequenceNumber = A[i]; // 현재 수열의 수
            if (currSequenceNumber >= num) { //현재 수열 값 >= 오름차순 자연수 : 값이 같아 질 때까지 push()수행
                while (currSequenceNumber >= num) { // push()
                    stack.push(num++);
                    stringBuffer.append("+\n");
                }
                stack.pop();
                stringBuffer.append("-\n");
            }
            else {  //현재 수열 값 < 오름차순 자연수: pop()을 수행하여 수열 원소를 꺼냅니다
                int n = stack.pop();
                // 스택의 가장 위의 수가 만들어야 하는 수열의 수 보다 크다면 수열 출력 불가능
                if (n > currSequenceNumber) {
                    System.out.println("NO");
                    result = false;
                    break;
                } //
                else {
                    stringBuffer.append("-\n");
                }
            }
        }
        if (result) System.out.println(stringBuffer);
    }
}
```
