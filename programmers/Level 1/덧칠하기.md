# 덧칠하기
* 시간 : 5분

```java
import java.util.Arrays;

class Solution {
    //n = 벽의 길이
    //m = 롤러 길이
    //section = 칠해야하는 구역
    public int solution(int n, int m, int[] section) {
        int answer = 0;
        boolean[] visited = new boolean[n];
        Arrays.fill(visited, true);
        for (int i : section) {
            visited[i - 1] = false;
        }

        for (int i = 0; i < n; i++) {
            if(!visited[i]) {
                for (int j = i; j < i + m; j++) {
                    if(j < visited.length) visited[j] = true;
                }
                answer++;
            }
        }

        return answer;
    }
}
```
