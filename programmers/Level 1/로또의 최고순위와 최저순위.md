# 로또의 최고순위와 최저순위
* 시간 : 7분

```java
import java.util.Arrays;
import java.util.HashSet;
import java.util.Set;

class Solution {
    public int[] solution(int[] lottos, int[] win_nums) {
        Set<Integer> winning = new HashSet<>();

        Arrays.stream(win_nums).forEach(winning::add);

        int count = (int) Arrays.stream(lottos).filter(winning::contains).count();
        int zeroCount = (int) Arrays.stream(lottos).filter(a -> a == 0).count();


        return new int[] {rank(count + zeroCount), rank(count)};
    }

    private int rank(int count){
        return switch (count) {
            case 6 -> 1;
            case 5 -> 2;
            case 4 -> 3;
            case 3 -> 4;
            case 2 -> 5;
            default -> 6;
        };
    }
}
```
